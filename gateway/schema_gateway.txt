CREATE TABLE `schema_gateway`.`air_conditioner` (`
ac_id` INT NOT NULL ,
`set_temp` REAL NOT NULL ,
`state` INT NOT NULL ,
`control_mode` INT NOT NULL ,
`time` INT NOT NULL ,
PRIMARY KEY (`ac_id`)) ENGINE = InnoDB;

CREATE TABLE `schema_gateway`.`fan` (
`fan_id` INT NOT NULL ,
`set_speed` REAL NOT NULL ,
`control_mode` INT NOT NULL ,
`set_time` INT NOT NULL ,
`time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
PRIMARY KEY (`fan_id`)) ENGINE = InnoDB;

CREATE TABLE `schema_gateway`.`energy_measure` (
`em_id` INT NOT NULL ,
`voltage` REAL NOT NULL ,
`current` REAL NOT NULL ,
`frequency` INT NOT NULL ,
`active_power` REAL NOT NULL ,
`power_factor` REAL NOT NULL ,
`time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ) ENGINE = InnoDB;

CREATE TABLE `schema_gateway`.`sensor_node` (
`sensor_id` INT NOT NULL ,
`temp` REAL NOT NULL ,
`humid` REAL NOT NULL ,
`wind` REAL NOT NULL ,
`pm25` INT NOT NULL ,
`time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ) ENGINE = InnoDB;

CREATE TABLE `schema_gateway`.`pmv_table` (
`id` INT NOT NULL ,
`met` REAL NOT NULL ,
`clo` INT NOT NULL ,
`pmv_ref` INT NOT NULL ,
PRIMARY KEY (`id`)) ENGINE = InnoDB;

CREATE TABLE `schema_gateway`.`registration_ac` (
`ac_id` INT NOT NULL ,
`sensor_link` TEXT NOT NULL ,
`model` INT NOT NULL ,
PRIMARY KEY (`ac_id`)) ENGINE = InnoDB;

CREATE TABLE `schema_gateway`.`registration_fan` (
`fan_id` INT NOT NULL ,
